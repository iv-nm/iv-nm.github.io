As for modes, I decided to make methods that will open app:

class AppLauncher:
    @staticmethod
    def open_brave():
        """ Open Brave """
        try:
            subprocess.Popen(["open", "-a", "Brave"])
            print("Visual Studio Code is opened.")
        except Exception as e:
            print(f"Failed to open Visual Studio Code: {e}")


    @staticmethod
    def open_vs_code():
        """ Open Visual Studio Code """
        try:
            subprocess.Popen(["code"])  # 'code' is the command to open VS Code
            print("Visual Studio Code is opened.")
        except Exception as e:
            print(f"Failed to open Visual Studio Code: {e}")

    @staticmethod
    def open_spotify():
        """ Open Yandex Music in the default web browser """
        try:
            subprocess.Popen(["open", "-a", "Spotify"])
            print("Spotify is opened.")
        except Exception as e:
            print(f"Failed to open Spotify: {e}")
    #etc.
